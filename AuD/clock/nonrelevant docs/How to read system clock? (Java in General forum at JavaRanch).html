import java.util.*;
class A {
   public static void main(String[] args) throws Exception {
      Calendar clock1 = Calendar.getInstance();
      long time1 = clock1.getTimeInMillis();
      Thread.sleep(1234);
      Calendar clock2 = Calendar.getInstance();
      long time2 = clock2.getTimeInMillis();
      System.out.println("time1="+time1);
      System.out.println("time2="+time2);
   }
}
java A
time1=1030748711075
time2=1030748712310

System.currentTimeMillis(); might be a little easier to use. For more accurate timing see http://www.boulder.nist.gov/timefreq/service/its.htm
